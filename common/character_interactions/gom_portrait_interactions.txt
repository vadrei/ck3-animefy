gom_hide_portrait_interaction = {
	category = interaction_category_religion
	desc = gom_hide_portrait_interaction_desc
	common_interaction = no
	use_diplomatic_range = no
	is_shown = {
		scope:recipient = {
			NOT = {
				has_trait = gom_portrait
			}
		}
	}

	is_valid = {
		
	}

	auto_accept = yes
	on_auto_accept = {
		scope:recipient = {
			add_trait = gom_portrait
		}
	}
}
